36def soil_carbon_aggregation(soil): 

37 """ 

38 Description: 

39 This function does the aggregation of carbon into pools 

40 "pseudocode_soil" S.6.C 

41 

42 Args: 

43 soil 

44 """ 

45 

46 for layer in soil.soil_layers: 

47 # soil mass calculations 

48 # S.6.D.1 

49 depth = layer.bottom_depth - soil.curr_layer_depth 

50 soil.curr_layer_depth += depth 

51 

52 soil_volume = depth * 10 * soil.area 

53 soil_mass = (layer.bulk_density / 0.001) * soil_volume 

54 

55 # aggregate Soil Carbon Pools (Active, Slow, and Passive) and CO2 Gas Flux. 

56 

57 # carbon pool conversion to percentages and their aggregation (i.e., Total %C) 

58 # S.6.D.2 

59 C_active_percent = (layer.C_active / soil_mass) 

60 C_slow_percent = (layer.C_slow / soil_mass) 

61 C_passive_percent = (layer.C_passive / soil_mass) 

62 

63 # S.6.D.3 

64 layer.C_percent = C_active_percent + C_slow_percent + C_passive_percent 

65 layer.org_C = layer.C_percent 

66 

67 # total soil carbon 

68 # S.6.D.4 

69 layer.C = layer.C_active + layer.C_slow + layer.C_passive 

70 layer.C_mg = layer.C * 0.001 

71 layer.C_g = layer.C * 0.1 

72 

73 # total CO2 Gas Aggregation 

74 # S.6.D.5 

75 AG_CO2_loss = layer.AG_met_to_C_active_loss + layer.AG_struct_to_C_active_loss + \ 

76 layer.AG_struct_to_C_slow_loss 

77 BG_CO2_loss = layer.BG_met_to_C_active_loss + layer.BG_struct_to_C_active_loss + \ 

78 layer.BG_struct_to_C_slow_loss 

79 

80 # S.6.D.6 

81 layer.C_CO2_loss_decomp = layer.C_active_loss + layer.C_slow_loss + layer.C_passive_loss 

82 

83 # S.6.D.7 

84 layer.C_CO2_loss = AG_CO2_loss + BG_CO2_loss + layer.C_CO2_loss_decomp 

85 

86 soil.curr_layer_depth = 0
