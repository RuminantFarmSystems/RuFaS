1""" 

2RUFAS: Ruminant Farm Systems Model 

3File name: lactating_cow_manure_excretion.py 

4Description: Determines manure excretion with information from the ration 

5 formulation, outputs used by the manure module. 

6Author(s): Militsa Sotirova, militsasotirova@gmail.com 

7 Joseph Merhi, jm2257@cornell.edu 

8""" 

9import math 

10from typing import Tuple 

11 

12from RUFAS.general_constants import GeneralConstants 

13from RUFAS.routines.animal.manure.general_manure import AnimalManureExcretions 

14from RUFAS.routines.animal.manure.general_manure import calculate_phosphorus_excretion_values 

15from RUFAS.routines.animal.ration.ration_driver import ration_report

18def manure_calculations(ration_formulation, 

19 feed, 

20 body_weight: float, 

21 days_in_milk: int, 

22 milk_protein: float, 

23 daily_milk_production: float, 

24 fecal_phosphorus: float, 

25 urine_phosphorus_required: float, 

26 methane_model: str, 

27 milk_fat: float, 

28 metabolizable_energy_intake: float) \ 

29 -> Tuple[float, AnimalManureExcretions]: 

30 """Calculates the manure excretion values for a cow with information from the ration formulation. 

31 

32 Parameters 

33 ---------- 

34 ration_formulation : Dict[str, float] 

35 Dictionary that stores the calculated ration. 

36 feed : Dict[str, float] 

37 A Feed object that contains information about the available feeds. 

38 body_weight : float 

39 Body weight of the current cow, kg. 

40 days_in_milk : int 

41 Days in milk, days. 

42 milk_protein : float 

43 Milk protein (from animal input), % of milk. 

44 daily_milk_production : float 

45 Daily milk production of the current cow, kg. 

46 fecal_phosphorus : float 

47 Amount of fecal phosphorus excreted by the current cow, g. 

48 urine_phosphorus_required : float 

49 Amount of phosphorus required for urine production, g. 

50 methane_model : str 

51 Methane model used for methane emission calculations, including Mutian, Mills, IPCC. 

52 milk_fat : float 

53 Milk fat (from animal input), % of milk. 

54 metabolizable_energy_intake : float 

55 Metabolizable energy intake, Mcal/kg dry matter. 

56 

57 Returns 

58 ------- 

59 float 

60 Total amount of phosphorus excreted by the given animal, g. 

61 AnimalManureExcretions 

62 A dictionary that contains the manure excretion values as specified 

63 in the AnimalManureExcretions class definition. 

64 

65 """ 

66 nutrient_amounts, nutrient_concentrations = ration_report( 

67 ration_formulation, feed.available_feeds) 

68 dry_matter_intake = nutrient_amounts['dm'] 

69 ASH_diet_content = nutrient_amounts['ash'] 

70 ASH_concentration = nutrient_concentrations["ash"] 

71 dry_matter_concentration = nutrient_concentrations['dm'] 

72 ADF_concentration = nutrient_concentrations['ADF'] 

73 CP_concentration = nutrient_concentrations['CP'] 

74 lignin_concentration = nutrient_concentrations['lignin'] 

75 NDF_concentration = nutrient_concentrations['NDF'] 

76 potassium_concentration = nutrient_concentrations['potassium'] 

77 EE_concentration = nutrient_concentrations["EE"] 

78 starch_concentration = nutrient_concentrations['starch'] 

79 

80 # Fecal water, kg [A.3E.A.1] 

81 fecal_water = (1.987 * dry_matter_intake 

82 + 0.348 * ADF_concentration 

83 - 0.412 * CP_concentration 

84 - 0.074 * dry_matter_concentration 

85 - 0.0057 * days_in_milk) 

86 

87 # Total Solids, kg [A.3E.A.2] 

88 # The amount of fecal solids is assumed to be equivalent to the amount of total solids 

89 fecal_solids = (-0.576 

90 + 0.370 * dry_matter_intake 

91 - 0.075 * CP_concentration 

92 + 0.059 * ADF_concentration) 

93 

94 # Total urine, kg [A.3E.A.3] 

95 urine = (-7.742 

96 + 0.388 * dry_matter_intake 

97 + 0.726 * CP_concentration 

98 + 2.066 * milk_protein) 

99 

100 # Manure excretion 

101 # Amount of feces and urine excreted daily by the growing heifer, kg [A.3E.A.4] 

102 total_manure_excreted = fecal_water + fecal_solids + urine 

103 

104 # Total manure nitrogen, kg [A.3E.B.1] 

105 manure_nitrogen = (20.3 

106 + 0.654 * (dry_matter_intake * GeneralConstants.KG_TO_GRAMS) * 

107 (CP_concentration * GeneralConstants.PROTEIN_TO_NITROGEN) / 100 

108 ) * GeneralConstants.GRAMS_TO_KG 

109 

110 # Urine nitrogen, kg [A.3E.B.2] 

111 urine_nitrogen = (12.0 

112 + 0.333 * (dry_matter_intake * GeneralConstants.KG_TO_GRAMS) * 

113 (CP_concentration * GeneralConstants.PROTEIN_TO_NITROGEN) / 100 

114 ) * GeneralConstants.GRAMS_TO_KG 

115 

116 # Fecal nitrogen, kg [A.3B.B.3] 

117 fecal_nitrogen = manure_nitrogen - urine_nitrogen 

118 

119 # Organic matter intake, kg [A.2.A.3] 

120 organic_matter_intake = dry_matter_intake - ASH_diet_content 

121 

122 # Degradable volatile solids, kg [A.3E.A.5] 

123 degradable_volatile_solids = (-1.017 

124 + 0.364 * organic_matter_intake 

125 + 0.029 * NDF_concentration 

126 - 0.023 * CP_concentration 

127 ) 

128 

129 # Total volatile solids, kg [A.3E.A.6] 

130 total_volatile_solids = (-1.201 

131 + 0.402 * organic_matter_intake 

132 + 0.036 * NDF_concentration 

133 - 0.024 * CP_concentration 

134 ) 

135 

136 # Non-degradable volatile solids, kg [A.3A.A.6] 

137 non_degradable_volatile_solids = total_volatile_solids - degradable_volatile_solids 

138 

139 # Urinary N concentration, g N/kg [A.3G.B.1] 

140 urinary_nitrogen_concentration = ( 

141 urine_nitrogen * GeneralConstants.KG_TO_GRAMS) / urine 

142 # Nitrogen concentration in urinary urea, g urea-N/L [A.3G.B.2] 

143 urine_urea_nitrogen_concentration = -1.16 + \ 

144 0.86 * urinary_nitrogen_concentration 

145 

146 # Clamp the urine urea nitrogen concentration to be between 2 and 12 g urea-N/L 

147 urine_urea_nitrogen_concentration_lower_bound = 2 

148 urine_urea_nitrogen_concentration_upper_bound = 12 

149 urine_urea_nitrogen_concentration = max(urine_urea_nitrogen_concentration_lower_bound, min( 

150 urine_urea_nitrogen_concentration, urine_urea_nitrogen_concentration_upper_bound)) 

151 

152 # Total ammoniacal nitrogen in the slurry top layer as a percentage of UUC, %, [A.3G.B.3] 

153 tan_percent_of_urea = 48.2 - 2.9 * urine_urea_nitrogen_concentration 

154 # Total ammoniacal nitrogen concentration in the manure slurry, 

155 # g ammoniacal nitrogen/L manure slurry [A.3G.B.4] 

156 total_ammoniacal_nitrogen_concentration = ( 

157 tan_percent_of_urea / 100) * urine_urea_nitrogen_concentration 

158 

159 # Amount of potassium excreted, g [A.3E.B.3] 

160 potassium = 7.21 * dry_matter_intake + 15944 * \ 

161 potassium_concentration / 100 - 164.5 

162 

163 # Methane Emissions 

164 methane_emission = 0.0 

165 if methane_model == "Mutian": # [A.3E.C.1] 

166 methane_emission = (- 126 

167 + 11.3 * dry_matter_intake 

168 + 2.30 * NDF_concentration 

169 + 28.8 * milk_fat 

170 + 0.148 * body_weight) 

171 

172 elif methane_model == "Mills": # [A.3E.C.2] 

173 starch_to_ADF_concentration_ratio = -0.0011 * \ 

174 starch_concentration / ADF_concentration 

175 temp = -(starch_to_ADF_concentration_ratio + 0.0045) * \ 

176 metabolizable_energy_intake * 4.184 

177 methane_emission = 45.98 * (1 - math.exp(temp)) / 0.05565 

178 

179 elif methane_model == "IPCC": # IPCC 

180 # Calculating gross energy concentration (Moraes et al. 2014) 

181 soluble_residue = 100 - ASH_concentration - \ 

182 NDF_concentration - CP_concentration - EE_concentration 

183 gross_energy_concentration = (0.263 * CP_concentration 

184 + 0.522 * EE_concentration 

185 + 0.198 * NDF_concentration 

186 + 0.160 * soluble_residue) # [A.3B.C.2] 

187 methane_emission = 0.065 * gross_energy_concentration * \ 

188 dry_matter_intake / 0.05565 # [A.3B.C.3] 

189 

190 

191 phosphorus_excretion_values = calculate_phosphorus_excretion_values( 

192 daily_milk_production=daily_milk_production, 

193 total_manure_excreted=total_manure_excreted, 

194 fecal_phosphorus=fecal_phosphorus, 

195 urine_phosphorus_required=urine_phosphorus_required 

196 ) 

197 

198 (total_phosphorus_excreted, inorganic_phosphorus_fraction, organic_phosphorus_fraction, 

199 manure_phosphorus_excreted, manure_phosphorus_fraction) = phosphorus_excretion_values 

200 

201 manure_excretion_values = AnimalManureExcretions( 

202 urea=urine_urea_nitrogen_concentration, 

203 urine=urine, 

204 total_ammoniacal_nitrogen_concentration=total_ammoniacal_nitrogen_concentration, 

205 urine_nitrogen=urine_nitrogen, 

206 manure_nitrogen=manure_nitrogen, 

207 manure_mass=total_manure_excreted, 

208 total_solids=fecal_solids, 

209 degradable_volatile_solids=degradable_volatile_solids, 

210 non_degradable_volatile_solids=non_degradable_volatile_solids, 

211 inorganic_phosphorus_fraction=inorganic_phosphorus_fraction, 

212 organic_phosphorus_fraction=organic_phosphorus_fraction, 

213 phosphorus=manure_phosphorus_excreted, 

214 phosphorus_fraction=manure_phosphorus_fraction, 

215 potassium=potassium, 

216 methane=methane_emission 

217 ) 

218 

219 return total_phosphorus_excreted, manure_excretion_values
